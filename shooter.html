const canvas = document.getElementById("game"); const ctx = canvas.getContext("2d"); canvas.width = window.innerWidth; canvas.height = window.innerHeight; const player = { x: canvas.width / 2 - 25, y: canvas.height - 80, w: 50, h: 50, speed: 7 }; let bullets = []; let enemies = []; let score = 0; function shoot() { bullets.push({ x: player.x + player.w / 2 - 3, y: player.y, w: 6, h: 15 }); } setInterval(() => { enemies.push({ x: Math.random() * (canvas.width - 40), y: -40, w: 40, h: 40, speed: 2 + Math.random() * 3 }); }, 1000); document.getElementById("left").ontouchstart = () => player.x -= 30; document.getElementById("right").ontouchstart = () => player.x += 30; document.getElementById("fire").ontouchstart = shoot; function update() { bullets.forEach((b, i) => { b.y -= 10; if (b.y < 0) bullets.splice(i, 1); }); enemies.forEach((e, i) => { e.y += e.speed; bullets.forEach((b, j) => { if ( b.x < e.x + e.w && b.x + b.w > e.x && b.y < e.y + e.h && b.y + b.h > e.y ) { enemies.splice(i, 1); bullets.splice(j, 1); score++; } }); if (e.y > canvas.height) enemies.splice(i, 1); }); ctx.clearRect(0, 0, canvas.width, canvas.height); ctx.fillStyle = "cyan"; ctx.fillRect(player.x, player.y, player.w, player.h); ctx.fillStyle = "yellow"; bullets.forEach(b => ctx.fillRect(b.x, b.y, b.w, b.h)); ctx.fillStyle = "red"; enemies.forEach(e => ctx.fillRect(e.x, e.y, e.w, e.h)); ctx.fillStyle = "white"; ctx.font = "24px Arial"; ctx.fillText("Score: " + score, 20, 35); requestAnimationFrame(update); } update();
